💡 Consultas SQL Útiles

-- Ver todos los nodos
SELECT * FROM cluster_nodes;

-- Contar métricas totales
SELECT COUNT(*) FROM metrics;

-- Últimas 20 métricas
SELECT * FROM metrics ORDER BY timestamp DESC LIMIT 20;

-- Promedio de CPU por nodo
SELECT host_id, AVG(cpu_usage) as avg_cpu 
FROM metrics 
GROUP BY host_id;

-- Métricas de las últimas 24 horas
SELECT * FROM metrics 
WHERE timestamp > NOW() - INTERVAL '24 hours';
